バックエンド（FastAPI）で実装する主なAPIエンドポイントを記述する。

\subsection{認証 (Auth)}

\textbf{ログイン} \\
POST /auth/login

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & email, password \\
\hline
\textbf{Response} & access\_token, token\_type, user\_role \\
\hline
\end{tabular}
\end{table}

\textbf{新規登録} \\
POST /auth/register

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & email, password, user\_type, profile\_data(name, address, etc.) \\
\hline
\textbf{Response} & user\_id, message \\
\hline
\end{tabular}
\end{table}

\subsection{商品 (Products)}

\textbf{店舗別商品一覧取得} \\
GET /stores/\{store\_id\}/products

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & store\_id (path) \\
\hline
\textbf{Response} & list[product\_object] \\
\hline
\end{tabular}
\end{table}

\textbf{商品登録（店舗のみ）} \\
POST /products

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & token, name, price, description, image, stock\_status \\
\hline
\textbf{Response} & product\_id, message \\
\hline
\end{tabular}
\end{table}
