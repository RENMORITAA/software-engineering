バックエンド（FastAPI）で実装する主なAPIエンドポイントを記述する。

\subsection{認証 (Auth)}

\textbf{ログイン} \\
POST /auth/login

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & email (str), password (str) \\
\hline
\textbf{Response} & access\_token (str), token\_type (str), user\_role (str) \\
\hline
\end{tabular}
\end{table}

\textbf{新規登録} \\
POST /auth/register

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & email (str), password (str), user\_type (str), profile\_data (dict) \\
\hline
\textbf{Response} & user\_id (int), message (str) \\
\hline
\end{tabular}
\end{table}

\subsection{商品 (Products)}

\textbf{店舗別商品一覧取得} \\
GET /stores/\{store\_id\}/products

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & store\_id (int) \\
\hline
\textbf{Response} & list[product\_object] \\
\hline
\end{tabular}
\end{table}

\textbf{商品登録（店舗のみ）} \\
POST /products

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Request} & token (str), name (str), price (int), description (str), image (str), stock\_status (bool) \\
\hline
\textbf{Response} & product\_id (int), message (str) \\
\hline
\end{tabular}
\end{table}

\subsection{注文 (Orders)}

\textbf{注文作成} \\
POST /orders

\begin{table}[h]
\centering

\hline
\textbf{Response} & order\_id (int), message (str) \\
\hline
\end{tabular}
\end{table}

PUT /orders/\{order\_id\}/status

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline

\textbf{Response} & message (str) \\
\hline
\end{tabular}
\end{table}
